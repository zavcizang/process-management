05/14 · INNOVATION #1
COW COPY-ON-WRITE
创新点一 · Copy-On-Write

COW 写时复制

# kernel/memory_manager.py
 
def fork_page_table(self, parent_pt):
"""fork 时复制页表（零拷贝）"""
child_pt = PageTable()
for page, entry in parent_pt.entries:
child_pt.map(page, entry.frame_id)
entry.is_writable = False # 只读
self._fm.incref(entry.frame_id)
return child_pt
 
def on_page_write(self, pt, page_num):
"""写入时触发 COW"""
entry = pt.get_entry(page_num)
if entry.is_writable: return False
# ref_count > 1 → 复制物理帧
new_frame = self._fm.allocate()
self._fm.copy_frame(entry.frame_id, new_frame)
entry.frame_id = new_frame
entry.is_writable = True
HOW IT WORKS
fork 前
父进程拥有独立页表
物理帧 ref_count = 1
fork 后
子进程页表指向同一物理帧
ref_count = 2 · 零拷贝 · 零额外内存
写入后
触发 COW → 只复制被修改的那一帧
其余帧继续共享 · 节省内存
"fork 时零内存拷贝，只有写入时才复制。" — Linux Kernel Design Pattern

# kernel/scheduler.py
BIG_STRIDE = 1_000_000
 
def register(self, pid, priority=128):
stride = 1_000_000 / (256 - priority)
self._registry[pid] = {
'stride': stride, 'pass_value': 0.0,
}
 
def dequeue(self):
return heapq.heappop(self._heap)
 
def update_after_run(self, pid, ticks):
entry['pass_value'] += entry['stride'] * ticks

08/14 · INNOVATION #3
PIPE IPC

管道进程间通信

# kernel/ipc.py
class Pipe:
BUFFER_SIZE = 4096
 
def write(self, data: bytes) -> int:
"""写入管道缓冲区"""
for byte in data:
self._buffer.append(byte)
return len(data)
 
def read(self, size: int) -> bytes:
"""从管道读取"""
result = bytearray()
while self._buffer and len(result) < size:
result.append(self._buffer.popleft())
return bytes(result)
